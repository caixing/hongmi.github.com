C++中的数组引用作为参数有两种  
1. 


    void print(int (&arr)[10]);
    int a[3],b[10];
    print(a);//error
    print(b);//right


2.  


    template <class T, size_t N> void print(T (&arr)[N]);
    print(a);//right
    print(b);//right
